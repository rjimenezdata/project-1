monthly_measles_summary()

monthly_measles_summary(): when given the measles dataframe, country name and a given year, it outputs the total number of measles cases for each month of that year.

Note: the country name can be spelled changing special characters (i.e. Turkiye)

  • summary using existing columns produced with custom function
monthly_measles_summary <- function(df, country_name, year_input) {
  
  # input checks
  required_cols <- c("Date", "Country", "Total Measles", "ASCII")
  if (!all(required_cols %in% names(df))) {
    missing_cols <- required_cols[!required_cols %in% names(df)]
    stop(paste("Missing required columns:", paste(missing_cols, 
                                                  collapse = ", ")))}
  
  if (!is.character(country_name) || length(country_name) != 1) {
    stop("Country Name must be a valid country.")}
  
  if (!is.numeric(year_input) || length(year_input) != 1) {
    stop("Year input must be a valid year.")}
  
  # empty dataset
  if (nrow(df) == 0) {
    warning("No data found.")}
  
  # handle special chars
  country_name_ascii <- stri_trans_general(country_name, "Latin-ASCII")
  
  # find the original country spelling from the dataset
  original_name <- df |>
    filter(ASCII == country_name_ascii) |>
    pull(Country) |>
    unique()
  
  # create summary table
  df |>
    mutate(Year = year(Date),
           Month = month(Date, label = TRUE, abbr = FALSE)) |>
    filter(ASCII == country_name_ascii,
           Year == year_input) |>
    group_by(Month) |>
    summarise("Total Measles Cases" = sum(`Total Measles`, na.rm = TRUE),
              .groups = "drop") |>
    arrange(Month) |>
    gt() |>
    tab_header(title = md(paste0("Monthly Measles Cases in ", original_name, 
                                " for the Year ", year_input))) |>
    cols_label(Month = md("**Month**"),
               "Total Measles Cases" = md("**Total Measles Cases**"))}

yoy_change <- function(df, country_name) {
  
  # input checks
  required_cols <- c("Year", "Country", "Total Measles per Million", "ASCII")
  
  if (!all(required_cols %in% names(df))) {
    missing_cols <- required_cols[!required_cols %in% names(df)]
    stop(paste("Missing required columns:", paste(missing_cols, collapse = ", ")))}
  
  if (!is.character(country_name) || length(country_name) != 1) {
    stop("Country Name must be a valid country.")}
  
  # empty dataset
  if (nrow(df) == 0) {

  original_name <- df |>
    filter(ASCII == country_name_ascii) |>
    pull(Country) |>
    unique()
  
  # create summary table
  df |>
    filter(ASCII == country_name_ascii) |>
    arrange(Year) |> 
    mutate(`Annual Change in Measles Rate` = `Total Measles per Million` -

summ_df()

summ_df(): when given the measles year dataframe, the function returns a dataframe summary that takes specific columns and lists the IQR, mean, median, and standard deviation of numeric variables and the unique number of levels for categorical variables.

  • summary using existing columns produced with custom function and iteration
summ_stats <- function(x){
  if (is.numeric(x)){
    tibble(Mean = round(mean(x, na.rm = T), 4),
           Median = round(median(x, na.rm = T), 4),
           `Standard Deviation` = round(sd(x, na.rm = T), 4),
           IQR = round(IQR(x, na.rm = T), 4))}
  
  else if (is.factor(x) || is.character(x)){
    tibble(Levels = length(unique(x)))}
  
  else {
    warning("Unsupported input type")
    return(NULL)}}

summ_df <- function(df){
  if (is.data.frame(df)){
    df |>
      mutate(Year = as.factor(Year)) |>
      map(summ_stats) |>
      bind_rows(.id = "Variable") |>
      pivot_longer(cols = !Variable,
                   names_to = "Summary") |>
      pivot_wider(names_from = Variable) |>
      select(Summary, Country, Year, `Total Pop.`, 
             `Total Suspected Cases`, `Total Measles per Million`) |>
      gt() |>
      tab_header(title = md("**Data Frame Summary**")) |>
      cols_label(Summary = md("**Summary**"),
                 Country = md("**Country**"),
                 Year = md("**Year**"),
                 `Total Pop.` = md("**Total Population**"),
                 `Total Suspected Cases` = md("**Total Suspected Measles and Rubella Cases**"),
                 `Total Measles per Million` = md("**Measles Rate Per Million**"))}
  else {
    warning("Input type not supported")
    return(NULL)}}

### summ_df()

**summ_df():** when given the measles year dataframe, the function returns a dataframe summary that takes specific columns and lists the IQR, mean, median, and standard deviation of numeric variables and the unique number of levels for categorical variables.

-   summary using existing columns produced with custom function and iteration

```{r}
summ_stats <- function(x){
  if (is.numeric(x)){
    tibble(Mean = round(mean(x, na.rm = T), 4),
           Median = round(median(x, na.rm = T), 4),
           `Standard Deviation` = round(sd(x, na.rm = T), 4),
           IQR = round(IQR(x, na.rm = T), 4))}
  
  else if (is.factor(x) || is.character(x)){
    tibble(Levels = length(unique(x)))}
  
  else {
    warning("Unsupported input type")
    return(NULL)}}

summ_df <- function(df){
  if (is.data.frame(df)){
    df |>
      mutate(Year = as.factor(Year)) |>
      map(summ_stats) |>
      bind_rows(.id = "Variable") |>
      pivot_longer(cols = !Variable,
                   names_to = "Summary") |>
      pivot_wider(names_from = Variable) |>
      select(Summary, Country, Year, `Total Pop.`, 
             `Total Suspected Cases`, `Total Measles per Million`) |>
      gt() |>
      tab_header(title = md("**Data Frame Summary**")) |>
      cols_label(Summary = md("**Summary**"),
                 Country = md("**Country**"),
                 Year = md("**Year**"),
                 `Total Pop.` = md("**Total Population**"),
                 `Total Suspected Cases` = md("**Total Suspected Measles and Rubella Cases**"),
                 `Total Measles per Million` = md("**Measles Rate Per Million**"))}
  else {
    warning("Input type not supported")
    return(NULL)}}

summ_df(measles_year)
```
